export function computeWeightTrend(records, extraWeightSources = []) {
  // Normaliser les sources supplementaires en pseudo-records HealthRecord
  const extraWeights = (extraWeightSources || [])
    .filter(s => s.weight_kg && s.date && isValidDate(s.date))
    .map(s => ({ type: "weight", value: s.weight_kg, date: s.date, id: s._source_id || s.id }));

  // Merger avec les HealthRecord weights existants, deduplication par date (HealthRecord prioritaire)
  const hrWeights = (records || [])
    .filter(r => r.type === "weight" && r.value && isValidDate(r.date));

  const hrDates = new Set(hrWeights.map(r => r.date));
  const mergedExtras = extraWeights.filter(w => !hrDates.has(w.date));

  const weights = [...hrWeights, ...mergedExtras]
    .sort((a, b) => new Date(b.date) - new Date(a.date));

          base44.entities.HealthRecord.filter({ dog_id: dog.id }).catch(() => []),
