**Etape 2 : Modifier BehaviorProgramCard pour utiliser completed_days**

Remplacer la barre de progression basee sur `elapsed` par une basee sur les jours completes, exactement comme TrainingCard :

```js
function BehaviorProgramCard({ program }) {
  const [open, setOpen] = useState(false);
  const [marking, setMarking] = useState(false);
  const elapsed = getElapsedDays(program.start_date);

  if (elapsed < 0 || elapsed >= 7) return null;
  const dayIndex = Math.min(elapsed, 6);
  const day = program.days?.[dayIndex];
  if (!day) return null;

  // Completion tracking (comme TrainingCard)
  const completedDays = program._completedDays || [];
  const completedCount = completedDays.length;
  const progress = Math.round((completedCount / 7) * 100);
  const todayDone = completedDays.includes(`d${elapsed}`);

  const handleMarkDone = async () => {
    if (todayDone || !program._bookmarkId) return;
    setMarking(true);
    try {
      const newCompleted = [...completedDays, `d${elapsed}`];
      await base44.entities.Bookmark.update(program._bookmarkId, { completed_days: newCompleted });
      // Optimistic: force re-render via parent callback (onLogged pattern)
      // Note: pas de callback disponible ici — l'etat sera mis a jour au prochain refresh de la page Home
    } catch (e) {
      console.warn("Mark done failed:", e?.message);
    }
    setMarking(false);
  };
```
